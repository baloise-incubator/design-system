import { Meta, Description, DocsContainer } from '@storybook/addon-docs'
import { BackToTop, TableOfContents } from 'hirsch-storybook-docs-toc'

import text_intro from './getting-started_intro.md'
import text_angular from './getting-started_angular.md'
import text_react from './getting-started_react.md'
import text_html from './getting-started_html.md'
import text_vue from './getting-started_vue.md'

<Meta
  anchor
  title="Development/Getting Started"
  parameters={{
    previewTabs: {
      canvas: { hidden: true },
    },
    docs: {
      container: ({ context }) => {
        const framework = context && context.globals && context.globals.framework
        let markdown = text_angular
        if (framework === 'react') {
          markdown = text_react
        }
        if (framework === 'html') {
          markdown = text_html
        }
        if (framework === 'vue') {
          markdown = text_vue
        }
        return (
          <DocsContainer context={context}>
            <TableOfContents />
            <div className="my-doc-content">
              {<Description markdown={text_intro} />}
              {<Description markdown={markdown} />}
            </div>
            <BackToTop />
          </DocsContainer>
        )
      },
    },
  }}
/>
